        days<-weekdays(as.Date(data[,"date"]))
        for(i in 1:length(days)){
                if(days[i] == "Sunday" | days[i] == "Saturday") {
                        days[i]<-"Weekend"     
                } else {
                        days[i]<-"Weekday"     
                }
        }

        data<-cbind(data, days)

        library(lattice)

        xyplot(steps~interval|days, data,type = "l", layout=c(1,2))
